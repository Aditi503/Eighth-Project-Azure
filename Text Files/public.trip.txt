IF NOT EXISTS (SELECT * FROM sys.external_file_formats WHERE name = 'SynapseDelimitedTextFormat') 
	CREATE EXTERNAL FILE FORMAT [SynapseDelimitedTextFormat] 
	WITH ( FORMAT_TYPE = DELIMITEDTEXT ,
	       FORMAT_OPTIONS (
			 FIELD_TERMINATOR = ',',
			 USE_TYPE_DEFAULT = FALSE
			))
GO

IF NOT EXISTS (SELECT * FROM sys.external_data_sources WHERE name = 'eighthprojectfilesystem_eighthprojectaccount_dfs_core_windows_net') 
	CREATE EXTERNAL DATA SOURCE [eighthprojectfilesystem_eighthprojectaccount_dfs_core_windows_net] 
	WITH (
		LOCATION = 'abfss://eighthprojectfilesystem@eighthprojectaccount.dfs.core.windows.net', 
		TYPE = HADOOP 
	)
GO

CREATE EXTERNAL TABLE dbo.Staging_Trips (
	[Trip_id] nvarchar(4000),
	[Rideable_type] nvarchar(4000),
	[Start_at] nvarchar(4000),
	[Ended_at] nvarchar(4000),
	[Start_station_id] nvarchar(4000),
	[End_station_id] nvarchar(4000),
	[Rider_id] nvarchar(4000)
	)
	WITH (
	LOCATION = 'Staging_Trips.csv',
	DATA_SOURCE = [eighthprojectfilesystem_eighthprojectaccount_dfs_core_windows_net],
	FILE_FORMAT = [SynapseDelimitedTextFormat]
	)
GO


SELECT TOP 100 * FROM dbo.Staging_Trips
GO
